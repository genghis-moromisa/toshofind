DB_PATH = os.path.join(os.path.dirname(__file__), "library.db")

app = Flask(__name__)

 # ------- ISBN正規化 -------

def normalize_isbn(raw: str | None) -> str | None:
    if not raw:
        return None
    s = raw.strip().upper()
    # ハイフン・空白などを除去（数字とXのみ残す）
    s = re.sub(r"[^0-9X]", "", s)
    if not s:
        return None
    return s


# ---------- DB ----------
def get_db():
    if "db" not in g:
        g.db = sqlite3.connect(DB_PATH)
        g.db.row_factory = sqlite3.Row
    return g.db

@app.teardown_appcontext
def close_db(_exc):
    db = g.pop("db", None)
    if db is not None:
        db.close()

def init_db():
    db = sqlite3.connect(DB_PATH)
    db.execute("""
    CREATE TABLE IF NOT EXISTS users (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      username TEXT UNIQUE NOT NULL,
      password_hash TEXT NOT NULL
    );
    """)
    db.execute("""
    CREATE TABLE IF NOT EXISTS books (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      isbn TEXT,
      title TEXT NOT NULL,
      authors TEXT,
      tags TEXT,
      location TEXT,
      notes TEXT,
      created_at TEXT DEFAULT CURRENT_TIMESTAMP,
      updated_at TEXT DEFAULT CURRENT_TIMESTAMP
    );
    """)
    db.commit()
    db.close()


def migrate_db():
    db = sqlite3.connect(DB_PATH)
    try:
        db.execute("ALTER TABLE books ADD COLUMN cover_url TEXT;")
    except Exception:
        pass
    try:
        db.execute("ALTER TABLE books ADD COLUMN source TEXT;")
    except Exception:
        pass
    try:
        db.execute("ALTER TABLE books ADD COLUMN meta_json TEXT;")
    except Exception:
        pass
    db.commit()
    db.close()


def ensure_admin():
    """初回起動で admin / admin を作る（必ず後で変更）"""
    db = sqlite3.connect(DB_PATH)
    db.row_factory = sqlite3.Row
    row = db.execute("SELECT id FROM users WHERE username = ?", ("admin",)).fetchone()
    if row is None:
        db.execute(
            "INSERT INTO users (username, password_hash) VALUES (?, ?)",
            ("admin", generate_password_hash("admin"))
        )
        db.commit()
    db.close()


# ---------- Auth + CSRF ----------
def login_required(fn):
    @wraps(fn)
    def wrapper(*args, **kwargs):
        if not session.get("user_id"):
            return redirect(url_for("login", next=request.path))
        return fn(*args, **kwargs)
    return wrapper

def get_csrf():
    token = session.get("csrf_token")
    if not token:
        token = secrets.token_urlsafe(32)
        session["csrf_token"] = token
    return token

def require_csrf():
    token = session.get("csrf_token")
    sent = request.form.get("csrf_token")
    if not token or not sent or token != sent:
        abort(400, "CSRF token mismatch")


# ---------- Routes ----------
@app.route("/login", methods=["GET", "POST"])
def login():
    if request.method == "POST":
        username = request.form.get("username", "").strip()
        password = request.form.get("password", "")
        db = get_db()
        user = db.execute("SELECT * FROM users WHERE username = ?", (username,)).fetchone()
        if user and check_password_hash(user["password_hash"], password):
            session["user_id"] = user["id"]
            session["username"] = user["username"]
            get_csrf()
            nxt = request.args.get("next") or url_for("index")
            return redirect(nxt)
        flash("ログインに失敗しました。")
    return render_template("login.html", csrf_token=get_csrf())

@app.route("/logout", methods=["POST"])
@login_required
def logout():
    require_csrf()
    session.clear()
    return redirect(url_for("login"))

@app.route("/")
@login_required
def index():
    q = request.args.get("q", "").strip()
    db = get_db()
    if q:
        like = f"%{q}%"
        rows = db.execute("""
          SELECT * FROM books
          WHERE title LIKE ? OR authors LIKE ? OR tags LIKE ? OR isbn LIKE ?
          ORDER BY updated_at DESC
        """, (like, like, like, like)).fetchall()
    else:
        rows = db.execute("SELECT * FROM books ORDER BY updated_at DESC LIMIT 200").fetchall()
    return render_template("index.html", books=rows, q=q, csrf_token=get_csrf())

            request.form.get("tags", "").strip() or None,
            request.form.get("location", "").strip() or None,
            request.form.get("notes", "").strip() or None,

        ))

@app.route("/add", methods=["GET", "POST"])

def fetch_book_metadata(isbn_raw: str) -> dict | None:
    """
    返り値例:
      {
        "isbn": "978...",
        "title": "...",
        "authors": "A, B",
        "publisher": "...",
        "pub_year": "2020",
        "notes": "...(説明/サブタイトル等)",
        "cover_url": "https://...",
        "source": "openlibrary" or "googlebooks",
        "meta_json": "{...}"  # 生JSON文字列（保険）
      }
    """
    isbn = normalize_isbn(isbn_raw)
    if not isbn:
        return None

    # ---- 1) Open Library Books API（ISBNで引ける）----
    # docs: /api/books?bibkeys=ISBN:...&format=json&jscmd=data :contentReference[oaicite:1]{index=1}
    try:
        url = "https://openlibrary.org/api/books"
        params = {"bibkeys": f"ISBN:{isbn}", "format": "json", "jscmd": "data"}
        r = requests.get(url, params=params, timeout=8)
        if r.ok:
            data = r.json()
            key = f"ISBN:{isbn}"
            if key in data:
                d = data[key]
                title = d.get("title")
                authors = ", ".join([a.get("name","") for a in d.get("authors", []) if a.get("name")]) or None
                publishers = ", ".join([p.get("name","") for p in d.get("publishers", []) if p.get("name")]) or None
                year = d.get("publish_date")
                cover = None
                if isinstance(d.get("cover"), dict):
                    cover = d["cover"].get("large") or d["cover"].get("medium") or d["cover"].get("small")

                if title:
                    return {
                        "isbn": isbn,
                        "title": title,
                        "authors": authors,
                        "publisher": publishers,
                        "pub_year": year,
                        "notes": None,
                        "cover_url": cover,
                        "source": "openlibrary",
                        "meta_json": json.dumps(d, ensure_ascii=False),
                    }
    except Exception:
        pass

    # ---- 2) Google Books API（isbn検索）----
    # docs: volumes?q=isbn:XXXX :contentReference[oaicite:2]{index=2}
    try:
        url = "https://www.googleapis.com/books/v1/volumes"
        params = {"q": f"isbn:{isbn}"}
        r = requests.get(url, params=params, timeout=8)
        if r.ok:
            data = r.json()
            items = data.get("items") or []
            if items:
                v = items[0].get("volumeInfo", {})
                title = v.get("title")
                authors = ", ".join(v.get("authors") or []) or None
                publisher = v.get("publisher")
                year = v.get("publishedDate")
                desc = v.get("description")
                cover = None
                if isinstance(v.get("imageLinks"), dict):
                    cover = v["imageLinks"].get("thumbnail") or v["imageLinks"].get("smallThumbnail")

                if title:
                    return {
                        "isbn": isbn,
                        "title": title,
                        "authors": authors,
                        "publisher": publisher,
                        "pub_year": year,
                        "notes": desc,
                        "cover_url": cover,
                        "source": "googlebooks",
                        "meta_json": json.dumps(v, ensure_ascii=False),
                    }
    except Exception:
        pass

    return None


@app.route("/scan_auto_add")
@login_required
def scan_auto_add():
    code = (request.args.get("code") or "").strip()
    isbn = normalize_isbn(code)
    if not isbn:
        flash("コードが空、またはISBNとして解釈できません。")
        return redirect(url_for("scan"))

    db = get_db()
    # すでに登録済みなら詳細へ
    hit = db.execute("SELECT * FROM books WHERE isbn = ? ORDER BY updated_at DESC", (isbn,)).fetchone()
    if hit:
        return redirect(url_for("book", book_id=hit["id"]))

    meta = fetch_book_metadata(isbn)
    if not meta:
        flash("自動取得に失敗しました。手動追加してください。")
        return redirect(url_for("add", isbn=isbn))

    # 自動でINSERT → 詳細へ
    db.execute("""
      INSERT INTO books (isbn, title, authors, tags, location, notes, cover_url, source, meta_json)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
    """, (
        meta["isbn"],
        meta["title"],
        meta.get("authors"),
        None,
        None,
        meta.get("notes"),
        meta.get("cover_url"),
        meta.get("source"),
        meta.get("meta_json"),
    ))
    db.commit()
    new_id = db.execute("SELECT last_insert_rowid() AS id").fetchone()["id"]
    flash(f"自動取得して追加しました（source: {meta.get('source')}）。必要なら編集してください。")
    return redirect(url_for("book", book_id=new_id))
